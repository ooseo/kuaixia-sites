Nginx 配置
伪静态配置(默认的配置)
location / {
	if (!-e $request_filename) {
        rewrite ^(.*)$ /index.php/$1 last;
        break;
	}
}

当注释 fastcgi_param PATH_INFO $path_info; 后的配置
location / {
	if (!-e $request_filename) {
        rewrite ^(.*)$ /index.php?s=$1 last;
        break;
	}
}

配置文件中删除其他静态文件配置
location ~ ^/alocal {
    root /www/wwwroot/www.yourdomain.com/kuaixia-sites;
    error_log /dev/null;
    access_log off;
    expires 30d;
}
